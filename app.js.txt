// FaceSwap Live - Main Application
class FaceSwapApp {
constructor() {
this.localStream = null;
this.peerConnection = null;
this.roomId = this.generateRoomId();
this.isCallActive = false;
this.currentFace = 'none';
this.customFaceImage = null;
    this.faceMesh = null;
    this.camera = null;
    this.canvas = document.getElementById('faceSwapCanvas');
    this.ctx = this.canvas.getContext('2d');
    
    this.fps = 0;
    this.lastTime = performance.now();
    this.frameCount = 0;
    
    this.init();
}

async init() {
    this.updateRoomId();
    await this.setupFaceMesh();
    await this.startCamera();
}

generateRoomId() {
    return Math.random().toString(36).substring(2, 8).toUpperCase();
}

updateRoomId() {
    document.getElementById('roomId').textContent = this.roomId;
}

async setupFaceMesh() {
    this.faceMesh = new FaceMesh({
        locateFile: (file) => {
            return `https://cdn.jsdelivr.net/npm/@mediapipe/face_mesh/${file}`;
        }
    });
    
    this.faceMesh.setOptions({
        maxNumFaces: 1,
        refineLandmarks: true,
        minDetectionConfidence: 0.5,
        minTrackingConfidence: 0.5
    });
    
    this.faceMesh.onResults(this.onFaceResults.bind(this));
}

async startCamera() {
    try {
        const video = document.getElementById('localVideo');
        const stream = await navigator.mediaDevices.getUserMedia({
            video: { width: { ideal: 1280 }, height: { ideal: 720 }, facingMode: 'user' },
            audio: true
        });
        
        this.localStream = stream;
        video.srcObject = stream;
        
        this.camera = new Camera(video, {
            onFrame: async () => {
                await this.faceMesh.send({ image: video });
            },
            width: 1280,
            height: 720
        });
        
        await this.camera.start();
        
        document.getElementById('status-indicator').style.background = '#10b981';
        document.getElementById('status-text').textContent = 'Camera Active';
        
    } catch (err) {
        console.error('Camera error:', err);
        document.getElementById('status-indicator').style.background = '#ef4444';
        document.getElementById('status-text').textContent = 'Camera Failed';
    }
}

onFaceResults(results) {
    this.canvas.width = results.image.width;
    this.canvas.height = results.image.height;
    
    this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
    this.ctx.drawImage(results.image, 0, 0);
    
    if (results.multiFaceLandmarks && results.multiFaceLandmarks.length > 0) {
        const landmarks = results.multiFaceLandmarks[0];
        document.getElementById('faceCount').textContent = results.multiFaceLandmarks.length;
        
        if (this.currentFace !== 'none') {
            this.applyFaceSwap(landmarks);
        }
        
        if (document.getElementById('smoothSkin').checked) {
            this.applySmoothSkin(landmarks);
        }
        
        if (document.getElementById('beautyFilter').checked) {
            this.applyBeautyFilter(landmarks);
        }
    } else {
        document.getElementById('faceCount').textContent = '0';
    }
    
    this.frameCount++;
    const currentTime = performance.now();
    if (currentTime - this.lastTime >= 1000) {
        this.fps = this.frameCount;
        this.frameCount = 0;
        this.lastTime = currentTime;
        document.getElementById('fps').textContent = this.fps;
    }
}

applyFaceSwap(landmarks) {
    let minX = 1, minY = 1, maxX = 0, maxY = 0;
    
    landmarks.forEach(lm => {
        minX = Math.min(minX, lm.x);
        minY = Math.min(minY, lm.y);
        maxX = Math.max(maxX, lm.x);
        maxY = Math.max(maxY, lm.y);
    });
    
    const x = minX * this.canvas.width;
    const y = minY * this.canvas.height;
    const w = (maxX - minX) * this.canvas.width;
    const h = (maxY - minY) * this.canvas.height;
    
    this.ctx.save();
    this.ctx.globalAlpha = 0.9;
    
    if (this.currentFace === 'celebrity1') {
        this.ctx.filter = 'sepia(0.3) saturate(1.2) hue-rotate(-10deg)';
    } else if (this.currentFace === 'celebrity2') {
        this.ctx.filter = 'sepia(0.2) saturate(0.8) hue-rotate(10deg) brightness(1.1)';
    } else if (this.currentFace === 'custom' && this.customFaceImage) {
        this.ctx.globalAlpha = 0.6;
        this.ctx.drawImage(this.customFaceImage, x, y, w, h);
    }
    
    this.ctx.beginPath();
    this.ctx.ellipse(x + w/2, y + h/2, w/2, h/2 * 1.2, 0, 0, Math.PI * 2);
    this.ctx.clip();
    this.ctx.drawImage(this.canvas, x, y, w, h, x, y, w, h);
    this.ctx.restore();
    
    this.ctx.strokeStyle = 'rgba(99, 102, 241, 0.3)';
    this.ctx.lineWidth = 2;
    this.ctx.beginPath();
    this.ctx.ellipse(x + w/2, y + h/2, w/2, h/2 * 1.2, 0, 0, Math.PI * 2);
    this.ctx.stroke();
}

applySmoothSkin(landmarks) {
    const faceRegion = this.getFaceRegion(landmarks);
    
    this.ctx.save();
    this.ctx.filter = 'blur(2px)';
    this.ctx.globalAlpha = 0.3;
    
    this.ctx.beginPath();
    const start = landmarks[0];
    this.ctx.moveTo(start.x * this.canvas.width, start.y * this.canvas.height);
    
    const outlineIndices = [10, 338, 297, 332, 284, 251, 389, 356, 454, 323, 361, 288, 397, 365, 379, 378, 400, 377, 152, 148, 176, 149, 150, 136, 172, 58, 132, 93, 234, 127, 162, 21, 54, 103, 67, 109];
    
    outlineIndices.forEach(idx => {
        const lm = landmarks[idx];
        this.ctx.lineTo(lm.x * this.canvas.width, lm.y * this.canvas.height);
    });
    
    this.ctx.closePath();
    this.ctx.clip();
    this.ctx.drawImage(this.canvas, 0, 0);
    this.ctx.restore();
}

applyBeautyFilter(landmarks) {
    const leftEye = landmarks[33];
    const rightEye = landmarks[263];
    
    [leftEye, rightEye].forEach(eye => {
        const x = eye.x * this.canvas.width;
        const y = eye.y * this.canvas.height;
        
        const gradient = this.ctx.createRadialGradient(x, y, 0, x, y, 30);
        gradient.addColorStop(0, 'rgba(255, 255, 255, 0.3)');
        gradient.addColorStop(1, 'rgba(255, 255, 255, 0)');
        
        this.ctx.fillStyle = gradient;
        this.ctx.beginPath();
        this.ctx.arc(x, y, 30, 0, Math.PI * 2);
        this.ctx.fill();
    });
}

getFaceRegion(landmarks) {
    let minX = 1, minY = 1, maxX = 0, maxY = 0;
    landmarks.forEach(lm => {
        minX = Math.min(minX, lm.x);
        minY = Math.min(minY, lm.y);
        maxX = Math.max(maxX, lm.x);
        maxY = Math.max(maxY, lm.y);
    });
    return {
        x: minX * this.canvas.width,
        y: minY * this.canvas.height,
        w: (maxX - minX) * this.canvas.width,
        h: (maxY - minY) * this.canvas.height
    };
}

selectFace(faceType) {
    this.currentFace = faceType;
    document.querySelectorAll('.face-option').forEach(el => {
        el.classList.remove('active');
        if (el.dataset.face === faceType) {
            el.classList.add('active');
        }
    });
}

async startCall() {
    this.peerConnection = new RTCPeerConnection({
        iceServers: [{ urls: 'stun:stun.l.google.com:19302' }]
    });
    
    this.localStream.getTracks().forEach(track => {
        this.peerConnection.addTrack(track, this.localStream);
    });
    
    this.peerConnection.ontrack = (event) => {
        const remoteVideo = document.getElementById('remoteVideo');
        remoteVideo.srcObject = event.streams[0];
        document.getElementById('remoteWrapper').style.display = 'block';
        document.getElementById('waitingState').style.display = 'none';
    };
    
    const offer = await this.peerConnection.createOffer();
    await this.peerConnection.setLocalDescription(offer);
    
    console.log('Call started, offer created');
    this.isCallActive = true;
    this.updateCallUI();
}

joinCall() {
    const roomId = prompt('Enter room ID:');
    if (roomId) {
        console.log('Joining room:', roomId);
        this.startCall();
    }
}

endCall() {
    if (this.peerConnection) {
        this.peerConnection.close();
        this.peerConnection = null;
    }
    
    document.getElementById('remoteWrapper').style.display = 'none';
    document.getElementById('waitingState').style.display = 'flex';
    
    this.isCallActive = false;
    this.updateCallUI();
}

updateCallUI() {
    const startBtn = document.getElementById('startCall');
    const joinBtn = document.getElementById('joinCall');
    const endBtn = document.getElementById('endCall');
    
    if (this.isCallActive) {
        startBtn.style.display = 'none';
        joinBtn.style.display = 'none';
        endBtn.style.display = 'block';
    } else {
        startBtn.style.display = 'block';
        joinBtn.style.display = 'block';
        endBtn.style.display = 'none';
    }
}

toggleMute() {
    const audioTrack = this.localStream.getAudioTracks()[0];
    if (audioTrack) {
        audioTrack.enabled = !audioTrack.enabled;
        document.getElementById('muteBtn').classList.toggle('active', !audioTrack.enabled);
    }
}

toggleVideo() {
    const videoTrack = this.localStream.getVideoTracks()[0];
    if (videoTrack) {
        videoTrack.enabled = !videoTrack.enabled;
        document.getElementById('videoBtn').classList.toggle('active', !videoTrack.enabled);
    }
}

}

document.addEventListener('DOMContentLoaded', () => {
app = new FaceSwapApp();
document.getElementById('faceUpload').addEventListener('change', (e) => {
    const file = e.target.files[0];
    if (file) {
        const reader = new FileReader();
        reader.onload = (event) => {
            const img = new Image();
            img.onload = () => {
                app.customFaceImage = img;
                app.selectFace('custom');
            };
            img.src = event.target.result;
        };
        reader.readAsDataURL(file);
    }
});

});
